            function execute({ items, cenikRows, division, threshold }) {
              var rows = cenikRows || [];
              var th = (typeof threshold === 'number') ? threshold : 0.8;

              // doba -> suffix SKU bundlu
              var termSuffix = { '1y': '-12', '3y': '-36', '5y': '-60' };
              // sluzba -> klicove slovo v popisu (spolehlivejsi nez cislo bundlu)
              var serviceKeyword = {
                'UTP': 'unified threat protection',
                'ENT': 'enterprise protection',
                'ATP': 'advanced threat protection'
              };

              function endsWithStr(s, suf) {
                s = String(s); suf = String(suf);
                return s.length >= suf.length && s.indexOf(suf, s.length - suf.length) !== -1;
              }

              var out = [];
              var anyLow = false;

              (items || []).forEach(function (it, idx) {
                var model   = String(it.model || '').toUpperCase().trim();
                var term    = String(it.term || '').toLowerCase().trim();
                var service = String(it.service || '').toUpperCase().trim();
                var qty     = (typeof it.qty === 'number') ? it.qty : (parseInt(it.qty, 10) || 1);

                // kandidati: stejna divize + SKU zacina na model
                var cand = rows.filter(function (r) {
                  var okDiv = !division || !r.division || String(r.division) === String(division);
                  return okDiv && String(r.sku || '').toUpperCase().indexOf(model) === 0;
                });

                // filtr podle sluzby (klicove slovo v popisu)
                if (service && serviceKeyword[service]) {
                  var kw = serviceKeyword[service];
                  cand = cand.filter(function (r) {
                    return String(r.description || '').toLowerCase().indexOf(kw) !== -1;
                  });
                }

                // filtr podle doby (suffix SKU)
                if (term && termSuffix[term]) {
                  var suf = termSuffix[term];
                  cand = cand.filter(function (r) { return endsWithStr(String(r.sku || ''), suf); });
                }

                // kdyz uzivatel nechtel zadnou sluzbu ani dobu -> hola HW polozka
                if (!service && !term) {
                  cand = cand.filter(function (r) {
                    return String(r.sku || '').toUpperCase() === model;
                  });
                }

                var chosen = null, confidence = 0, reason = '';
                if (cand.length === 1) {
                  chosen = cand[0]; confidence = 1;
                } else if (cand.length === 0) {
                  confidence = 0; reason = 'no_match';
                } else {
                  chosen = cand[0]; confidence = 0.5;
                  reason = 'multiple_candidates:' + cand.length;
                }
                if (confidence < th) anyLow = true;

                var cands = [];
                for (var c = 0; c < cand.length; c++) { cands.push(cand[c].sku); }

                out.push({
                  lineNo: idx + 1,
                  qty: qty,
                  requested: (it.model || '') + ' ' + (it.term || '') + ' ' + (it.service || ''),
                  sku: chosen ? chosen.sku : null,
                  description: chosen ? chosen.description : null,
                  cenikPrice: chosen ? chosen.price : null,
                  currency: chosen ? (chosen.currency || 'USD') : null,
                  confidence: confidence,
                  reason: reason,
                  candidates: cands
                });
              });

              return { matchedItems: out, hasLowConfidence: anyLow === true };
            }
          inputs: '{ items: $.data.requestItems, cenikRows: $.actions.readCenik.output.cenik, division: $.data.division, threshold: 0.8 }'

            function execute({ matchedItems, discountRows, partnerEmail, division }) {
              var items = matchedItems || [];
              var rows = discountRows || [];

              // najdi sazbu slevy pro partnera/divizi (0..1). Kdyz neni, rate=null.
              var rate = null;
              var dom = String(partnerEmail || '').split('@')[1] || '';
              for (var i = 0; i < rows.length; i++) {
                var r = rows[i];
                var okDiv = !division || !r.division || String(r.division) === String(division);
                var okPartner = (r.domain && dom && String(r.domain).toLowerCase() === dom.toLowerCase())
                             || (r.partner_email && String(r.partner_email).toLowerCase() === String(partnerEmail).toLowerCase());
                if (okDiv && okPartner && typeof r.rate === 'number') { rate = r.rate; break; }
              }

              function round2(n) { return Math.round(n * 100) / 100; }

              var out = [];
              items.forEach(function (it) {
                var listUnit = (typeof it.cenikPrice === 'number') ? it.cenikPrice : null;
                var listTotal = (listUnit !== null) ? round2(listUnit * it.qty) : null;
                var partnerUnit = (listUnit !== null && rate !== null) ? round2(listUnit * (1 - rate)) : null;
                var partnerTotal = (partnerUnit !== null) ? round2(partnerUnit * it.qty) : null;

                var copy = {};

                copy.listTotal = listTotal;
                copy.discountRate = rate;
                copy.partnerUnit = partnerUnit;
                copy.partnerTotal = partnerTotal;
                out.push(copy);
              });

              return { pricedItems: out, discountRate: rate, canPrice: rate !== null };
            }
